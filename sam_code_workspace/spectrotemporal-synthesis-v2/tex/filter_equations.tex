\begin{align*}
\shortintertext{Temporal impulse response:}
h(t) &= \sin(2 \pi t) t^{2} e^{-3.5 t}\\
\shortintertext{Spectral transfer function:}
h(f) &= \frac{f^2} {\mu^2} e^{(1 - \frac{f^2}{\mu^2}) }\\
&= \frac{f^2} {\mu^2} e^{\frac{(\mu^2 - f^2)} {\mu^2} }
\end{align*}
